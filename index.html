class LoveLetter:
    def __init__(self, name):
        self.name = name
        self.love_score = float('inf')  # because love is infinite

    def birthday_wish(self):
        print(f"Happy Birthday, {self.name}!")
        print("Today, I just want to say...")
        print("You're my favorite constant in this chaotic code.")
        print("You debug my worries and compile my dreams.")
        print("And yes, you're the only variable I’ll never update.")

    def love_status(self):
        return f"My love for you = {self.love_score}"

# Run the letter
if __name__ == "__main__":
    my_love = LoveLetter("Yauri")
    my_love.birthday_wish()
    print(my_love.love_status())
