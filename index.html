class LoveLetter:
    def __init__(self, name):
        self.name = name
        self.love_score = float('inf')  # because love is infinite 💖

    def birthday_wish(self):
        print(f"Happy Birthday, {self.name}!")
        print("Today, I just want to say...")
        print("You're my favorite constant in this ever-changing world.")
        print("You debug my worries and compile my happiness.")
        print("And yes, you're the only variable I never want to change.")

    def love_status(self):
        return f"My love for you = {self.love_score} ❤️"
